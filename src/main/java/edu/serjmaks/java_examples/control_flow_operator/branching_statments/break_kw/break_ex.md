## 01
```
// с меткой используется для выхода из вложенных циклов
// с меткой выполнение программы после оператора `break` начнется с места, расположенного сразу после этой метки
// "one:" - метка для оператора break
one:
for (int i = 1; i <= 3; i++) {
    System.out.println("iteration " + i + " :");
    for (int j = 1; j < 10; j++) {
        // при помощи ранее поставленной метки "one:" завершаем все циклы
        if (j > 5) break one;
        System.out.print(j + " ");
    }
    System.out.println("done");
}
```
